---
title: Floating Cards
description: Feature cards at different parallax speeds create a depth illusion — cards closer to you move faster, cards further away move slower.
navigation:
  icon: lucide:layout-grid
---

Apply `v-rellax` to each **column** rather than each individual card — both cards in a column travel together, giving a clean two-depth effect. The `center: true` option keeps each column anchored to its natural position at rest.

::prose-show-case

<script setup lang="ts">
  const columns = [
    {
      speed: -1,
      cards: [
        {
          icon: "lucide:zap",
          title: "v-rellax",
          desc: "Attach to any element in your template",
          accent: "text-violet-500",
          border: "border-violet-500/20",
          bg: "bg-violet-500/5",
        },
        {
          icon: "lucide:layers",
          title: "useRellaxGroup",
          desc: "Manage a whole collection at once",
          accent: "text-emerald-500",
          border: "border-emerald-500/20",
          bg: "bg-emerald-500/5",
        },
      ],
    },
    {
      speed: -3,
      cards: [
        {
          icon: "lucide:code-2",
          title: "useRellax",
          desc: "Full control from your script setup",
          accent: "text-blue-500",
          border: "border-blue-500/20",
          bg: "bg-blue-500/5",
        },
        {
          icon: "lucide:box",
          title: "RellaxElement",
          desc: "Typed props, reactive, exposes API",
          accent: "text-amber-500",
          border: "border-amber-500/20",
          bg: "bg-amber-500/5",
        },
      ],
    },
  ];

  const cardsScroll = useTemplateRef<HTMLElement>("cardsScroll");
  const cardsReady = ref(false);

  onMounted(async () => {
    await nextTick();
    cardsReady.value = true;
    await nextTick();
    if (cardsScroll.value) cardsScroll.value.scrollTop = 150;
  });
</script>

<template>
  <div class="w-full overflow-hidden rounded-xl border">
    <div
      class="bg-muted/60 text-muted-foreground flex items-center gap-2 border-b px-4 py-2 text-xs"

      Scroll this panel to watch columns float at different depths
    </div>

    <div ref="cardsScroll" class="bg-muted/10 relative h-[480px] overflow-y-auto">
      <div class="relative flex h-[900px] items-center justify-center px-6">
        <div v-if="cardsReady" class="flex w-full max-w-sm gap-4">
          <div
            v-for="col in columns"
            :key="col.speed"
            v-rellax="{
              speed: col.speed,
              wrapper: cardsScroll,
              center: true,
            }"
            class="flex flex-1 flex-col gap-4"
          >
            <div
              v-for="card in col.cards"
              :key="card.title"
              :class="`rounded-xl border p-4 shadow-sm ${card.border} ${card.bg}`"
            >
              <Icon :name="card.icon" :class="`size-5 ${card.accent}`" />
              <p class="text-foreground mt-3 text-sm font-semibold">{{ card.title }}</p>
              <p class="text-muted-foreground mt-1 text-xs leading-snug">{{ card.desc }}</p>
            </div>
          </div>
        </div>
      </div>
    </div>
  </div>
</template>

::

::prose-callout{variant="tip" title="center: true for card grids"}
`center: true` makes Rellax treat the element's natural scroll position as the "zero" point, so cards start where CSS placed them and drift symmetrically — perfect for grids and feeds.
::
